"""Helpers for doing HTTP using SSL using the Python stdlib.

The idea of the interfaces is that we can take "native" request objects and
return "native" response objects so user potentially has complete control.


ssl.SSLContext                              # SSLConfig

httplib.HTTPSConnection                     # Transfer
    request(method, url, body, headers)     # send
      calls:
        connect                                 # open connection
        putrequest                              # 1st after connect
        putheader                               # send one header
        endheaders                              # send blank line, opt. body
        send                                    # send whatever
        close                                   # close connection
    getresponse()                           # await response
    close()                                 # like canceling?

httplib.HTTPException

httplib.HTTPResponse                        # Response
    status                                  # ok, as code
    reason                                  # good solution
    read                                    # just use .body
    getheader, getheaders                   # just use .headers


httplib.HTTPMessage     Headers
"""

# For now, don't offer a choice, see if anything else is needed
_PROTOCOL_CHOICES = {
    # The option recommended in Python docs
    # "default": ssl.PROTOCOL_SSLv23,

    # same as PROTOCOL_TLS, deprecated in 3.6+
    # "ssl v2.3": ssl.PROTOCOL_SSLv23,

    # preferred alias only available in 3.6+
    # "tls": ssl.PROTOCOL_TLS,

    # deprecated options for interop
    # "ssl v2": ssl.PROTOCOL_SSLv2,
    # "ssl v3": ssl.PROTOCOL_SSLv3,
    # "tls v1": ssl.PROTOCOL_TLSv1,
    # "tls v1.1": ssl.PROTOCOL_TLSv1_1,
    # "tls v1.2": ssl.PROTOCOL_TLSv1_2,
}
# In 3.6+ the preferred name is PROTOCOL_TLS
if sys.version_info >= (3, 6):
    _PROTOCOL_CHOICES.update({
        "default": ssl.PROTOCOL_TLS,
    })
# In 2.7 to 3.5, the available name is PROTOCOL_SSLv23
else:
    _PROTOCOL_CHOICES.update({
        "default": ssl.PROTOCOL_SSLv23,
    })
