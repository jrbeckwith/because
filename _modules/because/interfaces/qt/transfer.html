[docs]def unpack_q_reply(q_reply):
    # TODO: shouldn't all of the below be encapsulated by Transfer?
    error = q_reply.errorString()
    status = q_reply.attribute(QNetworkRequest.HttpStatusCodeAttribute)
    redirect = q_reply.attribute(QNetworkRequest.RedirectionTargetAttribute)
    headers = q_reply.rawHeaderPairs()
    headers = [
        (key, value.data())
        for key, value in headers
    ]
    # Read out data.
    # TODO: This method isn't great for big bodies...
    data = q_reply.readAll()
    # Convert from QByteArray
    body = bytes(data)

    response = Response(
        error=error,
        redirect=redirect,
        status=status,
        headers=headers,
        body=iter(body),
    )
    return response

[docs]class Transfer(object):
    """Hold per-request state for one ongoing HTTP request cycle.

    This is also used as a device for bundling up multiple descriptions of a
    transfer's state to be handled by one callback, instead of requiring many
    separate callbacks to handle many different state notifications.
    """

    # n.b.: can't define signals here since this is not a QObject. The
    # workaround would be to have this instantiate a QObject subclass that
    # defines your signals. But the QNetworkReply already defines all the
    # signals we need here.

    log = LOG.getChild("Transfer")

    def __init__(
            self,
            request,
            on_finished=None,
            on_progress=None,
            timeout=None,
            send=None,
            log=None,
    ):
        # type: (Request, Any, Any, Any, Any, logging.Logger) -> None
        """
        :arg request:
            Request object describing the HTTP request to be performed.
        :arg on_finished:
            Callback for receiving notification that this transfer is over,
            whether successfully or unsuccessfully. Should take no arguments.
        :arg on_progress:
            Callback for receiving notification that this transfer has
            progressed. Must take one argument, being the transfer object that
            can be used to inspect progress if desired.
        :arg timeout:
            An integer representing the number of milliseconds to wait before
            canceling the whole transfer as "timed out."
        :arg log:
            logger instance to use for logging messages.
        :arg send:
            Callable which should be called, passing a QBuffer of body data, to
            send the request body.
        """
        # Store passed parameters
        self.request = request
        self._on_finished = on_finished
        # assert on_finished
        self._on_progress = on_progress
        self.timeout = timeout
        self._send = send
        self.log = log or self.log
        self._started = None
        self._stopped = None

        # Define attributes to hold underlying Qt objects
        self.q_reply = None
        self._q_array = None
        self._q_buffer = None
        self._q_timer = None
        # self._q_thread = 

        # For ensuring we only ran on_finished once. 
        self._ran_on_finished = False

        # What events do we need to handle, and which callbacks should we take?
        #
        # * We don't need much handling of the "finished" event because we
        #   have isRunning() and isFinished().
        #
        # * We don't need to handle "redirection" because we can use
        #   RedirectionTargetAttribute.
        #
        # * No known need for metaDataChanged.
        #
        # * No known need for SSL signals:
        #   encrypted, sslErrors, presharedKeyAuthenticationRequired.
        #
        # * Docs say that finished "will probably follow" signal error,
        #   but "probably" isn't really good enough, and it's otherwise unclear
        #   how to get the code instead of just errorString, so we want to
        #   handle that. But we can do that internally.
        #
        # * We need to handle uploadProgress and downloadProgress for UI,
        #   but it's a pain for users to define two different callbacks, when
        #   they can just define one that takes the transfer.
        #
        # * We need to handle internally generated timeouts, but it isn't clear
        #   that users should specify a separate callback for this, when it can
        #   reasonably be merged with whatever callback processes errors.
        #
        # * We can either use one big callback for all response-terminations,
        #   or we can separate out success and error streams. It's a tradeoff
        #   of whether our interface or their callback does the dispatch.
        #   but keeping them together makes it easy to use one namespace
        #   and one set of invariants regardless of how transfer ended.
        #
        # * Anything that doesn't vary frequently across transfers/sends is
        #   fair game for class changes rather than callback parameters.

        # These attributes collect state from uploadProgress and
        # downloadProgress signals, so that the user can give just one progress
        # callback instead of two with slightly different contracts.
        self.upload_sent = None
        self.upload_total = None
        self.download_received = None
        self.download_total = None

        # error
        # The kind of error is indicated per QNetworkReply::NetworkError, here
        # represented as a TransferError if not None.
        # (Not to be confused with an HTTP error, or an application-level error
        # message delivered over HTTP.)
        self.error = None

        # internally generated timeout from timer
        self.timed_out = False

        # TODO: also monitor for QNetworkReply::TimeoutError and
        # ProxyTimeoutError? only if these aren't actually http errors.

        self._start_timer()

        # TODO: UGH
        self.response = None

    def _start_timer(self):
        """Create and start a timer to enforce timeouts.
        """
        if self.timeout is not None:
            self.log.debug("Starting timer for +%sms", self.timeout)
            q_timer = QTimer()
            q_timer.timeout.connect(self.on_timeout)
            q_timer.start(self.timeout)
            self._q_timer = q_timer

    def _stop_timer(self):
        """Stop and remove any timer.
        """
        timer = self._q_timer
        if timer and timer.isActive():
            timer.stop()
        self._q_timer = None

    @property
    def upload_percent(self):
        """Property to compute percent completion of upload.

        This is based on uploadProgress.
        """
        if self.upload_total is None or self.upload_sent is None:
            return None
        elif self.upload_total == -1 or self.upload_sent == -1:
            return None
        elif self.upload_total == 0:
            return 100.0
        return (self.upload_sent / self.upload_total) * 100.0

    @property
    def upload_finished(self):
        """Property to check whether the response portion was completed.

        This is based on uploadProgress.
        """
        if all([
                self.upload_sent,
                self.upload_total,
                self.upload_sent == self.upload_total,
        ]):
            return True
        return False

    @property
    def download_percent(self):
        """Property to compute percent completion of upload.

        This is based on downloadProgress.
        """
        if self.download_total is None or self.download_received is None:
            return None
        elif self.download_total == -1 or self.download_received == -1:
            return None
        elif self.download_total == 0:
            return 100.0
        return (self.download_received / self.download_total) * 100.0

    @property
    def download_finished(self):
        """Property to check whether the response portion was completed.

        This is based on downloadProgress.
        """
        if all([
                self.download_received,
                self.download_total,
                self.download_received == self.download_total,
        ]):
            return True
        return False


[docs]    def on_progress(self):
        """Notify consumer of a progress update.

        Passes self to the callback so the consumer can inspect the transfer
        instance to see any status it finds pertinent.
        """
        if self._on_progress:
            self._on_progress(self)


    def _on_upload_progress(self, sent, total):
        """Handle uploadProgress from the QNetworkReply.
        """
        self.upload_sent = sent
        self.upload_total = total if total != -1 else None
        self.on_progress()

    def _on_download_progress(self, received, total):
        """Handle downloadProgress from the QNetworkReply.
        """
        self.download_received = received
        self.download_total = total if total != -1 else None
        self.on_progress()

        """Handle an error detected during response processing.

        error_code = code
        error_text = self.q_reply.errorString()
        self.error = TransferError(code=error_code, text=error_text)



[docs]    def on_timeout(self):
        """Handle the expiration of the deadline timer.
        """
        # Flag what happened
        self.timed_out = True

        # Time roughly when the request stopped for logging
        self._stopped = Datetime.utcnow()
        self.log.debug("Timing out after %s", self.elapsed)

        self._stop_timer()

        # Attempt to halt any ongoing activity
        if self.q_reply:
            self.q_reply.abort()


    @property
    def elapsed(self):
        if self._stopped:
            timedelta = self._stopped - self._started
        else:
            timedelta = Datetime.utcnow() - self._started
        return timedelta.total_seconds()


[docs]    def on_finished(self):
        """Handle completion of reply processing.

        This means that the end of the transfer is ready to be handled,
        response read, etc. whether or not there was an error.
        """
        td = Datetime.utcnow() - self._started
        # Ensure this only runs once.
        if self._ran_on_finished:
            self.log.debug("already ran _on_finished")
            return
        self._ran_on_finished = True
        self.log.debug("finished in {0}s".format(td.total_seconds()))
        self._stop_timer()
        if self._on_finished:
            self.log.debug("running _on_finished=%r", self._on_finished)
            self.response = unpack_q_reply(self.q_reply)
            self.log.debug("HTTP response\n{0}".format(self.response.pretty()))
            self._on_finished(self)
        else:
            self.log.debug("no _on_finished to run")

[docs]    def start(self):
        # type: () -> None

        self._started = Datetime.utcnow()

        request = self.request
        # Pack the request body for Qt.
        # * sendCustomRequest needs a QIODevice.
        # * We can get one using QBuffer.
        # * QBuffer needs a QByteArray.
        # * TODO: extra copy and double memory usage?
        # * TODO: can QBuffer parent solve some problems? maybe use q_request?
        # * TODO: can we just pass q_array intending const QByteArray &data? or
        #   even just a python bytes? possible??
        q_array = QByteArray(request.body) if request.body else QByteArray()
        q_buffer = QBuffer(q_array)

        # Call the function provided by Client to send the request body.
        self._started = Datetime.utcnow()
        q_reply = self._send(q_buffer)

        # Connect QNetworkReply signals directly to callbacks on self.
        self._connect(q_reply)

        # # TODO: it still feels stupid to call into a client method once with
        # # request, then do very little work, then call into it with other
        # # representations of same request
        # # q_request = self.client.q_request(self.request)
        # q_reply = self.client.send_q_request(
        #     self.request,
        #     # q_request,
        #     # self.request.method,
        #     q_buffer,
        # )

        # Hold references to underlying Qt objects involved in response read.
        # If there is no reference to the QBuffer after this method exits, QGIS
        # will crash in short order. If there is no reference to the QArray
        # after this method exits, the result will be flaky; it may work, or
        # silently fail, or crash QGIS.
        # Likely explanation: when the refcount goes to 0, Python collects the
        # object, and then Qt reads inappropriately from the freed pointer,
        # causing a SEGFAULT.
        # The read doesn't seem to be caused by the finished signal or any of
        # our callbacks, because the crash happens even if those ar eall
        # removed.
        # Ideally, sendCustomRequest would keep its own buffers alive long
        # enough to complete the send, and then Qt wouldn't read them after...
        self._q_buffer = q_buffer
        self._q_array = q_array


    # Kicking off the request requires access to the QNAM and SSL config.
    # To keep these encapsulated in client, those details are handled in
    # Client.send before this is called.
    # However, this means that connect has to be public API, passing
    # q_reply/q_array/q_buffer. :(
    # otherwise, client has to put per-transfer state/refs somewhere else.

    def _connect(self, q_reply):
        """Associate this transfer to the given QNetworkReply.

        This is for internal use because it exposes Qt API.
        """

        # Make this method idempotent; don't double up on signal subscriptions
        if self.q_reply == q_reply:
            return

        # A Transfer should map to a QNetworkReply 1:1, assuming it maps at
        # all. If this Transfer has already been connected to a QNetworkReply,
        # we don't want it handling signals from another QNetworkReply as well.
        # We could avoid that by disconnecting all the old signals, but it
        # would be a confusing contract violation to silently cause those old
        # signal subscriptions to stop firing.
        if self.q_reply is not None:
            raise TransferError(
                "Transfer instance is already associated with a QReply",
            )

        # Hang on to the QReply we just got
        # * so we can ensure this transfer instance isn't associated to another
        # * to ensure it's not collected prematurely, since nothing else has it
        self.q_reply = q_reply

        # In order of importance, for what little it's worth; since there is a
        # period after Qt request sent and before QReply signals are connected
        signal_mapping = [
            ("finished", "on_finished"),
            ("error", "on_error"),
            ("uploadProgress", "_on_upload_progress"),
            ("downloadProgress", "_on_download_progress"),
        ]
        for signal_name, method_name in signal_mapping:
            signal = getattr(q_reply, signal_name)
            method = getattr(self, method_name)
            signal.connect(method)
        self.log.debug("Connected signals")


[docs]    def close(self):
        self.log.debug("close")

        # Prevent timeout callback firing
        self._stop_timer()

        # If we close while still running, intent is abort.
        # Yes, this is different from the Qt API!
        if self.q_reply and self.q_reply.isRunning():
            self.q_reply.abort()

        # Prevent further callback access through self
        self._on_finished = None
        self._on_progress = None

        # Close the QBuffer
        if self._q_buffer is not None:
            self._q_buffer.close()

        # Empty the QByteArray to prevent further access
        if self._q_array is not None:
            self._q_array.truncate(0)

        # Prevent further reads from the QNetworkReply
        if self.q_reply:
            self.q_reply.close()

        # Break references to ensure Qt objects can be collected
        self._q_array = None
        self._q_buffer = None
        self._q_timer = None
        self.q_reply = None
