[docs]class TokenService(Service):
    """Define endpoints for BCS token service.
    """
    def __init__(self):
        # type: () -> None
        endpoints = {
            # send credentials, receive Auth0 token for headers
            u"token": Endpoint(
                path=u"/token/",
                methods=[u"POST"],
                parameters={
                    # body parameters currently not used...
                    # "username": Text,
                    # "password": Text,
                }

            ),
        }
        headers = Headers.merged(
            DEFAULT_HEADERS,
            Headers([
                (b"Content-Type", b"application/json"),
            ])
        )
        super(TokenService, self).__init__(
            endpoints,
            headers=headers,
        )


[docs]    def login_request(self, base, username, password, headers=None):

        body = json.dumps({
            u"username": username,
            u"password": password,
        }).encode("utf-8")

        return self.request(
            endpoint_name=u"token", method=u"POST", base=base, body=body,
            headers=headers,
        )



[docs]    def parse(self, response):
        # TODO: check HTTP status
        parsed = json.loads(response.body.decode("utf-8"))
        token = parsed.get("token")
        if not token:
            return
        token = token.encode("utf-8")
        return token



[docs]    def parse_metadata(self, token):
        parts = token.split(b".", 2)
        if len(parts) < 2:
            return
        middle = parts[1]
        padding = (len(middle) % 4) * b"="
        payload = base64.b64decode(middle + padding)
        data = json.loads(payload.decode("utf-8"))
        return data
