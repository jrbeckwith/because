[docs]class Basemap(object):
    """Describes a basemap result from the basemaps discovery endpoint.
    """
    # methods: always just GET.
    def __init__(
            self, path,
            # In basemaps service, kind of essential
            standard="XYZ",
            tile_format="PNG",
            # 
            title=None,
            attribution="Boundless",
            access_list=None,
            description="",
            thumbnail=None,
            headers=None,
    ):
        """
        :arg path:
            URI template to use. This will probably specify how to encode x, y,
            and z coordinates and any other parameters needed.
        :arg title:
            Title like "Boundless OSM Basemap" (aka 'name' in /basemaps result)
        """
        self.path = path
        self.standard = standard
        self.tile_format = tile_format
        self.title = title
        self.attribution = attribution
        self.access_list = access_list
        self.description = description
        self.thumbnail = thumbnail
        self.headers = headers

        # don't delete these untless implementation supports something else
        if self.tile_format != "PNG":
            raise InvalidBasemap(
                "only tile_format=PNG basemaps are supported"
            )
        if self.standard != "XYZ":
            raise InvalidBasemap(
                "only XYZ basemaps are supported"
            )

        parameter_names = re.findall("{([a-zA-Z_]+)}", path)
        expected = set(["x", "y", "z"])
        if set(parameter_names) != expected:
            raise InvalidBasemap(
                "expected parameters for x, y, and z"
            )

[docs]    def from_dict(cls, data):
        """Convert from dict representing JSON object from basemap discovery
        """
        attr_mapping = {
            "accessList": "access_list",
            "attribution": "attribution",
            "description": "description",
            "endpoint": "path",
            # reserve "name" for the parameter given after service, e.g.
            # /basemaps/boundless/{osm} or /basemaps/mapbox/{outdoors}
            "name": "title",
            "standard": "standard",
            "thumbnail": "thumbnail",
            "tileFormat": "tile_format",
        }
        remapped = {
            attr_mapping.get(source_attr, source_attr): data[source_attr]
            for source_attr in data.keys()
        }
        return Basemap(**remapped)
