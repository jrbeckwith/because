class _Finish(object):
    """Calls a callback
    """
    def __init__(self, finish, subscriber):
        # type: () -> None
        """
        :arg finish:
            Callable to call when Transfer is done.
        :arg subscriber:
            Another object which wants to know about transfer.
        """
        self.finish = finish
        self.subscriber = subscriber

    def on_start(self, obj):
        # type: (Any) -> None
        """
        """
        # Forward to whatever function caller wanted called
        if self.subscriber:
            self.subscriber.on_start(obj)

    def on_finish(self, response):
        # type: (Response) -> None
        """
        """
        self.finish(response)

        # Forward
        if self.subscriber:
            self.subscriber.on_finish(response)

    def __init__(
            self,
            interface="python",
            env="dev",
            ssl_config=None,
            log=None,
    ):
        # type: () -> None
        """
        """

        client_cls = INTERFACES.get(interface)
        if client_cls is None:
            raise InvalidObject("no known interface {0!r}".format(interface))

        self.host = HOSTS.get(env)
        if not self.host:
            raise InvalidObject("no known host for env {0!r}".format(env))

        self.client = client_cls(
            ssl_config=ssl_config,
            log=log,
        )

        # Instantiate all the services on top of our base URL

        self.token_service = TokenService()
        self.routing_service = RoutingService()
        self.geocoding_service = GeocodingService()
        self.basemaps_service = BasemapsService()
        self.search_service = SearchService()

        # Store token to use in Authorization headers
        self.token = None

        # Store roles?
        self.metadata = None
        self.roles = set([])

        # Cache basemaps from the enumeration endpoint.
        # Indexed by strings like "Mapbox Satellite"
        self.basemaps = {}


[docs]    def headers(self):
        if not self.token:
            raise NotLoggedIn()
        return Headers([
            (
                b"Authorization", "Bearer {0}".format(
                    self.token.decode("utf-8")
                ).encode("utf-8")
            ),
        ])


    def _go(self, request, finish, subscriber=None):
        # Frontend needs a result, but the result doesn't have to be to this
        # method. Which is lucky, since users of non-blocking implementations
        # like Qt will expect not to block, and this method can't just
        # "await" or "yield" without assumptions about caller interface that
        # aren't sustainable on Python 2. Callbacks will work fine anywhere.
        #
        # To abstract off that detail, we need a way to tell the implementation
        # to call a method on self whenever a response is ready.
        #
        # Non-blocking implementations like Qt also need an opportunity after
        # the request is composed and the transfer is started to do things like
        # subscribing to progress events. In Qt's case, this means it needs to
        # get at the QNetworkReply returned by QNAM.sendCustomRequest so that
        # it can connect its signals to slots. And it may also mean that it
        # needs to be notified when the transfer is done to clean up.
        #
        # So callers which are particular about such things can pass a
        # callback, which will be called with whatever implementation-specific
        # object the caller needs to peek at, and another callback which will
        # be called when the transfer is done. And these might as well get
        # stuck on an object because otherwise we're passing around tons of
        # loose functions.
        #
        # So we pass these callback bundles (call them subscribers) to the
        # implementation, and when transfer starts, the implementation calls
        # subscriber.on_start() with the native object, and when the transfer
        # ends, e.g. before transfer.wait() returns, the implementation calls
        # subscriber.on_finish() with the Response object, and on_finish can
        # also be used to call self._finish_login() here.
        #
        # To achieve that we make this proxy object that calls the specified
        # method when we're done, and otherwise forwards to subscriber.
        proxy = _Finish(finish, subscriber)

        # Have client decide how it wants to send that request
        transfer = self.client.send(request, subscriber=proxy)

        # Caller can wait on result
        return transfer


[docs]    def login(self, username, password, subscriber=None):
        # type: (Any, Any) -> None
        """Use the token service to grab an auth token.

        This should be run before other methods.
        """
        request = self.token_service.login_request(
            self.host.url, username, password,
        )
        return self._go(request, self._finish_login, subscriber=subscriber)


    def _finish_login(self, response):
        token = self.token_service.parse(response)
        self.token = token


[docs]    def get_basemaps(self, subscriber=None):
        """Use the basemaps service to enumerate available basemaps.

        For now this has to be called and waited on before doing basemap
        stuff... :(
        """
        if not self.token:
            raise NotLoggedIn()

        request = self.basemaps_service.request(
            u"metadata",
            method=u"GET",
            base=self.host.url,
            headers=self.headers(),
        )
        transfer = self.client.send(request, subscriber=subscriber)
        return Result(transfer, self.basemaps_service.parse)


    def _finish_basemaps(self, response):
        basemaps = self.basemaps_service.parse_basemaps(response)
        # Cache it for next time
        self.basemaps = basemaps


[docs]    def basemap(self, name):
        """Get the basemap of the specified name.

        If there is no such basemap, returns None.
        """
        if not self.token:
            raise NotLoggedIn()

        # TODO: bleh we need to chain futures to do this properly
        transfer = self.get_basemaps()
        transfer.wait()
        return self.basemaps.get(name.lower())



[docs]    def geocode(self, address, service="mapbox", subscriber=None):
        """Use the geocoding service to geocode an address.
        """
        if not self.token:
            raise NotLoggedIn()

        if not address:
            raise Exception("FIXME")

        request = self.geocoding_service.request(
            u"forward",
            method=u"GET",
            base=self.host.url,
            values={
                "service": service,
                "address": address,
            },
            headers=self.headers(),
        )
        transfer = self.client.send(request, subscriber=subscriber)
        return Result(transfer, self.geocoding_service.parse_forward)

[docs]    def route(self, origin, *waypoints, **kwargs):
        """Use the routing service to get a route through locations/addresses.
        """
        # These wouldn't be in **kwargs but Python 2 is lacking
        service = kwargs.get("service") or "mapbox"
        subscriber = kwargs.get("subscriber")

        if not self.token:
            raise NotLoggedIn()

        if not origin or not waypoints:
            raise Exception("FIXME")
        waypoints = "|".join([origin] + list(waypoints))

        request = self.routing_service.request(
            u"waypoints",
            method=u"GET",
            base=self.host.url,
            values={
                "service": service,
                "waypoints": waypoints,
            },
            headers=self.headers(),
        )
        transfer = self.client.send(request, subscriber=subscriber)
        return Result(transfer, self.routing_service.parse)
