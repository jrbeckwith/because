"""Helpers for doing HTTP using SSL.

This really only arises as a way of mitigating the huge number of parameters
that some HTTP APIs offer for customizing the behavior of TLS/SSL. Instead of
passing 43 arguments to Client, you can just pass whichever ones apply to your
implementation to your implementation's flavor of SSLConfig, and pass that to
the Client so I can pretend that Client is simpler than it is.
"""

[docs]class SSLKey(object):
    """TBD.
    """

    def __init__(
            self, data, 
            pass_phrase=b"", algorithm=u"dsa", encoding=u"pem",
            key_type=u"private",
    ):
        # type: (bytes, bytes, Text, Text, Text) -> None
        # Don't pass duck types that are likely to be misunderstood
        if not isinstance(data, bytes):
            raise InvalidSSLConfig(
                u"encoded key data must be bytes"
            )
        if not isinstance(pass_phrase, bytes):
            raise InvalidSSLConfig(
                u"pass phrase must be bytes"
            )

        self.data = data
        self.pass_phrase = pass_phrase
        self.algorithm = algorithm
        self.encoding = encoding
        self.key_type = key_type




[docs]class SSLConfig(object):
    """

    Automates QSslConfiguration.set* calls and ensure that they happen up
    front, which is important when some may have no effect later.

    Hides the dance of creating a QSslKey.

    Makes it easier to set SSL options, and sets defaults marked in Qt docs as
    more secure, so that less secure options must be explicitly specified.
    """
    # TODO:
    # setCaCertificates
    # setCiphers
    # setLocalCertificate
    # setDefaultConfiguration?

    default_ssl_options = {
        # Prevents BEAST, but "causes problems with a large number of servers"
        "empty_fragments": False,

        # Prevents injection of plaintext into SSL session
        "legacy_renegotiation": False,

        # Disable compression to prevent CRIME attack
        "compression": False,
    }

    def __init__(
            self,
            protocol=None,
            private_key=None,
            private_key_algorithm=None,
            private_key_encoding=None,
            private_key_pass_phrase=None,
            ssl_options=None,
    ):
        self.protocol = protocol or "secureprotocols"
        self.private_key = private_key
        self.private_key_algorithm = private_key_algorithm
        self.private_key_encoding = private_key_encoding
        self.private_key_pass_phrase = private_key_pass_phrase

        # Start from the defaults, then override them
        # This also ensures we are not munging the passed dict
        self.ssl_options = self.default_ssl_options.copy()
        if ssl_options:
            self.ssl_options.update(ssl_options)

        # These internal-use methods just populate q_ssl_config.
        self._set_ssl_protocol()
        self._set_private_key()
        self._set_ssl_options()

    def _set_private_key(self):
        # type: () -> None
        if self.private_key:
            ssl_key = SSLKey(
                data=self.private_key,
                algorithm=self.private_key_algorithm,
                encoding=self.private_key_encoding,
                pass_phrase=self.private_key_pass_phrase,
            )
